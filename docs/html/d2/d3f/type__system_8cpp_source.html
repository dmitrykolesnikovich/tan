9  auto *orig = val->getType();
10  auto *loaded = val;
11  if (is_lvalue) { /// IMPORTANT
12  loaded = cs->get_builder()->CreateLoad(val);
13  TAN_ASSERT(orig->getNumContainedTypes() == 1);
14  orig = orig->getContainedType(0); /// we only care the type of the rvalue of orig_val
15  }
16  bool is_pointer1 = orig->isPointerTy();
17  bool is_pointer2 = dest->isPointerTy();
18  size_t s1 = orig->getPrimitiveSizeInBits();
19  size_t s2 = dest->getPrimitiveSizeInBits();
20  /// early return if types are the same
21  if (is_llvm_type_same(orig, dest)) { return loaded; };
22  if (is_pointer1 && is_pointer2) {
23  /// cast between pointer types (including pointers to pointers)
24  return cs->get_builder()->CreateBitCast(loaded, dest);
25  } else if (orig->isIntegerTy() && dest->isIntegerTy() && s2 != 1) {
26  /// different int types except dest is bool (1-bit int)
27  if (s1 == s2) {
28  return loaded;
29  } else {
30  return cs->get_builder()->CreateZExtOrTrunc(loaded, dest);
31  }
32  } else if (orig->isIntegerTy() && dest->isFloatingPointTy()) { /// int to float/double
33  if (is_signed) {
34  return cs->get_builder()->CreateUIToFP(loaded, dest);
35  } else {
36  return cs->get_builder()->CreateSIToFP(loaded, dest);
37  }
38  } else if (orig->isFloatingPointTy() && dest->isIntegerTy()) { /// float/double to int
39  if (is_signed) {
40  return cs->get_builder()->CreateFPToUI(loaded, dest);
41  } else {
42  return cs->get_builder()->CreateFPToSI(loaded, dest);
43  }
44  } else if (orig->isFloatingPointTy() && dest->isFloatingPointTy()) { /// float <-> double
45  return cs->get_builder()->CreateFPCast(loaded, dest);
46  } else if (dest->isIntegerTy(1)) { /// all types to bool, equivalent to val != 0
47  if (orig->isFloatingPointTy()) {
48  return cs->get_builder()->CreateFCmpONE(loaded, ConstantFP::get(cs->get_builder()->getFloatTy(), 0.0f));
49  } else if (orig->isPointerTy()) {
50  s1 = cs->get_ptr_size();
51  loaded = cs->get_builder()->CreateIntToPtr(loaded, cs->get_builder()->getIntNTy((unsigned) s1));
52  return cs->get_builder()
53  ->CreateICmpNE(loaded, ConstantInt::get(cs->get_builder()->getIntNTy((unsigned) s1), 0, false));
54  } else {
55  return cs->get_builder()
56  ->CreateICmpNE(loaded, ConstantInt::get(cs->get_builder()->getIntNTy((unsigned) s1), 0, false));
57  }
58  } else {
59  // TODO: complex type
60  throw std::runtime_error("Invalid type conversion");
61  }
62 }

74 int TypeSystem::CanImplicitCast(ASTTyPtr t1, ASTTyPtr t2) {
75  TAN_ASSERT(t1);
76  TAN_ASSERT(t2);
77  if (t1.get() == t2.get()) { return 0; } /// since some ASTTy are cached, compare pointer and early return
78  if (*t1 == *t2) { return 0; }
79  size_t s1 = t1->get_size_bits();
80  size_t s2 = t2->get_size_bits();
81 
82  if (t1->is_bool()) {
83  return 0;
84  } else if (t1->is_bool()) {
85  return 1;
86  } else if (t1->is_int() && t2->is_int()) { /// between integers
87  /// should be both unsigned or both signed
88  if (t1->is_unsigned() ^ t2->is_unsigned()) { return -1; }
89  return s1 >= s2 ? 0 : 1;
90  } else if (t1->is_floating() && t2->is_int()) { /// float/double and int
91  return 0;
92  } else if (t1->is_int() && t2->is_floating()) { /// int and float/double
93  return 1;
94  } else if (t1->is_floating() && t2->is_floating()) { /// float/double and float/double
95  return s1 >= s2 ? 0 : 1;
96  }
97  // TODO: complex type
98  return -1;
99 }
