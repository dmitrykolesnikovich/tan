34  TAN_ASSERT(_resolved);
35  Ty base = TY_GET_BASE(_tyty);
36  Value *ret = nullptr;
37  Type *type = this->to_llvm_type(cs);
38  switch (base) {
39  case Ty::INT:
40  case Ty::CHAR:
41  case Ty::BOOL:
42  ret = ConstantInt::get(type, std::get<uint64_t>(_default_value));
43  break;
44  case Ty::FLOAT:
45  ret = ConstantFP::get(type, std::get<float>(_default_value));
46  break;
47  case Ty::DOUBLE:
48  ret = ConstantFP::get(type, std::get<double>(_default_value));
49  break;
50  case Ty::STRING:
51  ret = cs->get_builder()->CreateGlobalStringPtr(std::get<std::string>(_default_value));
52  break;
53  case Ty::VOID:
54  TAN_ASSERT(false);
55  break;
56  case Ty::STRUCT: {
57  auto st = ast_cast<ASTStruct>(cs->get(_type_name));
58  TAN_ASSERT(st);
59  ret = st->get_llvm_value(cs);
60  break;
61  }
62  case Ty::POINTER:
63  ret = ConstantPointerNull::get((PointerType *) type);
64  break;
65  case Ty::ARRAY: {
66  auto *e_type = _children[0]->to_llvm_type(cs);
67  ret = create_block_alloca(cs->get_builder()->GetInsertBlock(), e_type, _n_elements, "const_array");
68  for (size_t i = 0; i < _n_elements; ++i) {
69  auto *idx = cs->get_builder()->getInt32((unsigned) i);
70  auto *e_val = _children[i]->get_llvm_value(cs);
71  auto *e_ptr = cs->get_builder()->CreateGEP(ret, idx);
72  cs->get_builder()->CreateStore(e_val, e_ptr);
73  }
74  break;
75  }
76  default:
77  TAN_ASSERT(false);
78  }

207 void ASTTy::resolve() {
208  if (_resolved) { return; }
209  _ty = this->shared_from_this();
210  /// resolve children if they are ASTTy
211  for (auto c: _children) {
212  auto t = ast_cast<ASTTy>(c);
213  if (t && t->_type == ASTType::TY && !t->_resolved) { t->resolve(); }
214  }
215  auto *tm = Compiler::GetDefaultTargetMachine(); /// can't use _cs here, cuz some ty are created by ASTTy::Create()
216  Ty base = TY_GET_BASE(_tyty);
217  Ty qual = TY_GET_QUALIFIER(_tyty);
218  switch (base) {
219  case Ty::INT: {
220  _size_bits = 32;
221  _type_name = "i32";
222  _is_int = true;
223  _default_value.emplace<uint64_t>(0);
224  if (TY_IS(qual, Ty::BIT8)) {
225  _size_bits = 8;
226  _type_name = "i8";
227  } else if (TY_IS(qual, Ty::BIT16)) {
228  _size_bits = 16;
229  _type_name = "i16";
230  } else if (TY_IS(qual, Ty::BIT64)) {
231  _size_bits = 64;
232  _type_name = "i64";
233  }
234  if (TY_IS(qual, Ty::UNSIGNED)) {
235  _is_unsigned = true;
236  if (_size_bits == 8) {
237  _dwarf_encoding = llvm::dwarf::DW_ATE_unsigned_char;
238  } else {
239  _dwarf_encoding = llvm::dwarf::DW_ATE_unsigned;
240  }
241  } else {
242  if (_size_bits == 8) {
243  _dwarf_encoding = llvm::dwarf::DW_ATE_signed_char;
244  } else {
245  _dwarf_encoding = llvm::dwarf::DW_ATE_signed;
246  }
247  }
248  break;
249  }
250  case Ty::CHAR:
251  _type_name = "char";
252  _size_bits = 8;
253  _dwarf_encoding = llvm::dwarf::DW_ATE_unsigned_char;
254  _is_unsigned = true;
255  _default_value.emplace<uint64_t>(0);
256  _is_int = true;
257  break;
258  case Ty::BOOL:
259  _type_name = "bool";
260  _size_bits = 1;
261  _dwarf_encoding = llvm::dwarf::DW_ATE_boolean;
262  _default_value.emplace<uint64_t>(0);
263  _is_bool = true;
264  break;
265  case Ty::FLOAT:
266  _type_name = "float";
267  _size_bits = 32;
268  _dwarf_encoding = llvm::dwarf::DW_ATE_float;
269  _default_value.emplace<float>(0);
270  _is_float = true;
271  break;
272  case Ty::DOUBLE:
273  _type_name = "double";
274  _size_bits = 64;
275  _dwarf_encoding = llvm::dwarf::DW_ATE_float;
276  _default_value.emplace<double>(0);
277  _is_double = true;
278  break;
279  case Ty::STRING:
280  _type_name = "u8*";
281  _size_bits = tm->getPointerSizeInBits(0);
282  _default_value.emplace<std::string>("");
283  _align_bits = 8;
284  _is_ptr = true;
285  break;
286  case Ty::VOID:
287  _type_name = "void";
288  _size_bits = 0;
289  _dwarf_encoding = llvm::dwarf::DW_ATE_signed;
290  break;
291  case Ty::STRUCT:
292  // TODO: align size in bits
293  _align_bits = 64;
294  _is_struct = true;
295  break;
296  case Ty::ARRAY: {
297  auto et = ast_cast<ASTTy>(_children[0]);
298  auto s = ast_cast<ASTNumberLiteral>(_children[1]);
299  TAN_ASSERT(et);
300  TAN_ASSERT(s);
301  _n_elements = _children.size();
302  _type_name = "[" + et->get_type_name() + ", " + std::to_string(_n_elements) + "]";
303  _is_ptr = true;
304  _is_array = true;
305  _size_bits = tm->getPointerSizeInBits(0);
306  _align_bits = et->get_size_bits();
307  _dwarf_encoding = llvm::dwarf::DW_ATE_address;
308  break;
309  }
310  case Ty::POINTER: {
311  auto e = ast_cast<ASTTy>(_children[0]);
312  TAN_ASSERT(e);
313  e->resolve();
314  _type_name = e->get_type_name() + "*";
315  _size_bits = tm->getPointerSizeInBits(0);
316  _align_bits = e->get_size_bits();
317  _is_ptr = true;
318  _dwarf_encoding = llvm::dwarf::DW_ATE_address;
319  break;
320  }
321  default:
322  TAN_ASSERT(false);
323  }
324  _resolved = true;
325 }

329 size_t ASTTy::nud_array() {
330  _end_index = _start_index + 1; /// skip "["
331  ASTNodePtr element = nullptr;
332  /// element type
333  if (_parser->at(_end_index)->value == "]") { /// empty
334  report_code_error(_parser->at(_end_index), "The array type and size must be specified");
335  } else {
336  element = std::make_shared<ASTTy>(_parser->at(_end_index), _end_index);
337  _end_index = element->parse(_parser, _cs); /// this set the _type_name of child
338  }
339  _parser->peek(_end_index, TokenType::PUNCTUATION, ",");
340  ++_end_index; /// skip ","
341 
342  /// size
343  auto size = _parser->peek(_end_index);
344  if (size->_type != ASTType::NUM_LITERAL) {
345  report_code_error(_parser->at(_end_index), "Expect an unsigned integer");
346  }
347  auto size1 = ast_cast<ASTNumberLiteral>(size);
348  if (size1->is_float() || size1->_ivalue < 0) {
349  report_code_error(_parser->at(_end_index), "Expect an unsigned integer");
350  }
351  _n_elements = static_cast<size_t>(size1->_ivalue);
352  for (size_t i = 0; i < _n_elements; ++i) { _children.push_back(element->get_ty()); }
353  /// set _type_name to [<element type>, <n_elements>]
354  _type_name = "[" + _type_name + ", " + std::to_string(_n_elements) + "]";
355  ++_end_index; /// skip "]"
356  return _end_index;
357 }

359 size_t ASTTy::nud() {
360  _end_index = _start_index;
361  Token *token = nullptr;
362  while (!_parser->eof(_end_index)) {
363  token = _parser->at(_end_index);
364  if (basic_tys.find(token->value) != basic_tys.end()) { /// base types
365  _tyty = TY_OR(_tyty, basic_tys[token->value]);
366  } else if (qualifier_tys.find(token->value) != qualifier_tys.end()) { /// TODO: qualifiers
367  if (token->value == "*") { /// pointer
368  auto sub = std::make_shared<ASTTy>(token, _end_index + 1);
369  /// swap self and child
370  sub->_tyty = this->_tyty;
371  _tyty = Ty::POINTER;
372  _children.push_back(sub);
373  auto ei = sub->parse(_parser, _cs);
374  if (_end_index + 1 != ei) { _end_index = ei - 1; }
375  }
376  } else if (token->type == TokenType::ID) { /// struct or array
377  // TODO: identify type aliases
378  _type_name = token->value; /// _type_name is the name of the struct
379  _tyty = TY_OR(_tyty, Ty::STRUCT);
380  } else if (token->value == "[") {
381  _tyty = TY_OR(_tyty, Ty::ARRAY);
382  _end_index = nud_array(); /// set _type_name in nud_array()
383  } else { break; }
384  ++_end_index;
385  }
386  resolve(); /// fill in relevant member variables
387  return _end_index;
388 }
