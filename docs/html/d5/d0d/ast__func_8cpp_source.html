235 
236  /// arguments

245  ++_end_index;

247  }
248  }

250  ++_end_index;

310 ASTFunctionPtr ASTFunctionCall::get_callee() const {
311  if (!_callee) {
312  auto func_candidates = _cs->get_functions(_name);
313  /// always prefer the function with lowest cost if multiple candidates are callable
314  /// one implicit cast -> +1 cost
315  /// however, if two (or more) functions have the same score, an error is raise (ambiguous call)
316  size_t cost = (size_t) -1;
317  for (const auto &f : func_candidates) {
318  size_t n = f->get_n_args();
319  if (n != _children.size()) { continue; }
320  bool good = true;
321  size_t c = 0;
322  for (size_t i = 0; i < n; ++i) { /// check every argument (return type not checked)
323  auto arg = f->get_arg(i);
324  TAN_ASSERT(arg->is_typed());
325  auto actual_arg = _children[i];
326  /// allow implicit cast from actual_arg to arg, but not in reverse
327  auto t1 = arg->get_ty();
328  auto t2 = actual_arg->get_ty();
329  if (*t1 != *t2) {
330  if (0 != TypeSystem::CanImplicitCast(t1, t2)) {
331  good = false;
332  break;
333  }
334  ++c;
335  }
336  }
337  if (good) {
338  if (c < cost) {
339  _callee = f;
340  cost = c;
341  } else if (c == cost) { report_code_error(_token, "Ambiguous function call: " + _name); }
342  }
343  }
344  }
345  if (!_callee) { report_code_error(_token, "Unknown function call: " + _name); }
346  return _callee;
347 }
