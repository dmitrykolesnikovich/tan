263 // Parse the command line options as presented and return the operation
264 // specified. Process all modifiers and check to make sure that constraints on
265 // modifier/operation pairs have not been violated.
266 static ArchiveOperation parseCommandLine() {
267  if (MRI) {
268  if (!PositionalArgs.empty() || !Options.empty()) {
269  badUsage("cannot mix -M and other options");
270  }
271  runMRIScript();
272  }

274  // Keep track of number of operations. We can only specify one
275  // per execution.
276  unsigned NumOperations = 0;
277 
278  // Keep track of the number of positional modifiers (a,b,i). Only
279  // one can be specified.
280  unsigned NumPositional = 0;
281 
282  // Keep track of which operation was requested
283  ArchiveOperation Operation;
284 
285  bool MaybeJustCreateSymTab = false;
286 
287  for (unsigned i = 0; i < Options.size(); ++i) {
288  switch (Options[i]) {
289  case 'd':
290  ++NumOperations;
291  Operation = Delete;
292  break;
293  case 'm':
294  ++NumOperations;
295  Operation = Move;
296  break;
297  case 'p':
298  ++NumOperations;
299  Operation = Print;
300  break;
301  case 'q':
302  ++NumOperations;
303  Operation = QuickAppend;
304  break;
305  case 'r':
306  ++NumOperations;
307  Operation = ReplaceOrInsert;
308  break;
309  case 't':
310  ++NumOperations;
311  Operation = DisplayTable;
312  break;
313  case 'x':
314  ++NumOperations;
315  Operation = Extract;
316  break;
317  case 'c':
318  Create = true;
319  break;
320  case 'l': /* accepted but unused */
321  break;
322  case 'o':
323  OriginalDates = true;
324  break;
325  case 'O':
326  DisplayMemberOffsets = true;
327  break;
328  case 'P':
329  CompareFullPath = true;
330  break;
331  case 's':
332  Symtab = true;
333  MaybeJustCreateSymTab = true;
334  break;
335  case 'S':
336  Symtab = false;
337  break;
338  case 'u':
339  OnlyUpdate = true;
340  break;
341  case 'v':
342  Verbose = true;
343  break;
344  case 'a':
345  getRelPos();
346  AddAfter = true;
347  NumPositional++;
348  break;
349  case 'b':
350  getRelPos();
351  AddBefore = true;
352  NumPositional++;
353  break;
354  case 'i':
355  getRelPos();
356  AddBefore = true;
357  NumPositional++;
358  break;
359  case 'D':
360  Deterministic = true;
361  break;
362  case 'U':
363  Deterministic = false;
364  break;
365  case 'N':
366  getCountParam();
367  break;
368  case 'T':
369  Thin = true;
370  // Thin archives store path names, so P should be forced.
371  CompareFullPath = true;
372  break;
373  case 'L':
374  AddLibrary = true;
375  break;
376  case 'V':
377  cl::PrintVersionMessage();
378  exit(0);
379  case 'h':
380  printHelpMessage();
381  exit(0);
382  default:
383  badUsage(std::string("unknown option ") + Options[i]);
384  }
385  }
386 
387  // At this point, the next thing on the command line must be
388  // the archive name.
389  getArchive();
390 
391  // Everything on the command line at this point is a member.
392  Members.assign(PositionalArgs.begin(), PositionalArgs.end());
393 
394  if (NumOperations == 0 && MaybeJustCreateSymTab) {
395  NumOperations = 1;
396  Operation = CreateSymTab;
397  if (!Members.empty()) {
398  badUsage("the 's' operation takes only an archive as argument");
399  }
400  }
401 
402  // Perform various checks on the operation/modifier specification
403  // to make sure we are dealing with a legal request.
404  if (NumOperations == 0) {
405  badUsage("you must specify at least one of the operations");
406  }
407  if (NumOperations > 1) {
408  badUsage("only one operation may be specified");
409  }
410  if (NumPositional > 1) {
411  badUsage("you may only specify one of 'a', 'b', and 'i' modifiers");
412  }
413  if (AddAfter || AddBefore) {
414  if (Operation != Move && Operation != ReplaceOrInsert) {
415  badUsage("the 'a', 'b' and 'i' modifiers can only be specified with "
416  "the 'm' or 'r' operations");
417  }
418  }
419  if (CountParam) {
420  if (Operation != Extract && Operation != Delete) {
421  badUsage("the 'N' modifier can only be specified with the 'x' or 'd' "
422  "operations");
423  }
424  }
425  if (OriginalDates && Operation != Extract) {
426  badUsage("the 'o' modifier is only applicable to the 'x' operation");
427  }
428  if (OnlyUpdate && Operation != ReplaceOrInsert) {
429  badUsage("the 'u' modifier is only applicable to the 'r' operation");
430  }
431  if (AddLibrary && Operation != QuickAppend) {
432  badUsage("the 'L' modifier is only applicable to the 'q' operation");
433  }
434 
435  // Return the parsed operation to the caller
436  return Operation;
437 }

460 // Implement the 't' operation. This function prints out just
461 // the file names of each of the members. However, if verbose mode is requested
462 // ('v' modifier) then the file type, permission mode, user, group, size, and
463 // modification time are also printed.
464 static void doDisplayTable(StringRef Name, const object::Archive::Child &C) {
465  if (Verbose) {
466  Expected<sys::fs::perms> ModeOrErr = C.getAccessMode();
467  failIfError(ModeOrErr.takeError());
468  sys::fs::perms Mode = ModeOrErr.get();
469  printMode((Mode >> 6) & 007);
470  printMode((Mode >> 3) & 007);
471  printMode(Mode & 007);
472  Expected<unsigned> UIDOrErr = C.getUID();
473  failIfError(UIDOrErr.takeError());
474  outs() << ' ' << UIDOrErr.get();
475  Expected<unsigned> GIDOrErr = C.getGID();
476  failIfError(GIDOrErr.takeError());
477  outs() << '/' << GIDOrErr.get();
478  Expected<uint64_t> Size = C.getSize();
479  failIfError(Size.takeError());
480  outs() << ' ' << format("%6llu", Size.get());
481  auto ModTimeOrErr = C.getLastModified();
482  failIfError(ModTimeOrErr.takeError());
483  // Note: formatv() only handles the default TimePoint<>, which is in
484  // nanoseconds.
485  // TODO: fix format_provider<TimePoint<>> to allow other units.
486  sys::TimePoint<> ModTimeInNs = ModTimeOrErr.get();
487  outs() << ' ' << formatv("{0:%b %e %H:%M %Y}", ModTimeInNs);
488  outs() << ' ';
489  }
490 
491  if (C.getParent()->isThin()) {
492  if (!sys::path::is_absolute(Name)) {
493  StringRef ParentDir = sys::path::parent_path(ArchiveName);
494  if (!ParentDir.empty()) {
495  outs() << sys::path::convert_to_slash(ParentDir) << '/';
496  }
497  }
498  outs() << Name;
499  } else {
500  outs() << Name;
501  if (DisplayMemberOffsets) {
502  outs() << " 0x" << utohexstr(C.getDataOffset(), true);
503  }
504  }
505  outs() << '\n';
506 }

512 static bool comparePaths(StringRef Path1, StringRef Path2) {
513  // When on Windows this function calls CompareStringOrdinal
514  // as Windows file paths are case-insensitive.
515  // CompareStringOrdinal compares two Unicode strings for
516  // binary equivalence and allows for case insensitivity.
517  #ifdef _WIN32
518  SmallVector<wchar_t, 128> WPath1, WPath2;
519  failIfError(sys::windows::UTF8ToUTF16(normalizePath(Path1), WPath1));
520  failIfError(sys::windows::UTF8ToUTF16(normalizePath(Path2), WPath2));
521 
522  return CompareStringOrdinal(WPath1.data(), WPath1.size(), WPath2.data(),
523  WPath2.size(), true) == CSTR_EQUAL;
524  #else
525  return normalizePath(Path1) == normalizePath(Path2);
526  #endif
527 }

529 // Implement the 'x' operation. This function extracts files back to the file
530 // system.
531 static void doExtract(StringRef Name, const object::Archive::Child &C) {
532  // Retain the original mode.
533  Expected<sys::fs::perms> ModeOrErr = C.getAccessMode();
534  failIfError(ModeOrErr.takeError());
535  sys::fs::perms Mode = ModeOrErr.get();
536 
537  llvm::StringRef outputFilePath = sys::path::filename(Name);
538  if (Verbose) {
539  outs() << "x - " << outputFilePath << '\n';
540  }
541 
542  int FD;
543  failIfError(sys::fs::openFileForWrite(outputFilePath, FD, sys::fs::CD_CreateAlways, sys::fs::OF_None, Mode), Name);
544 
545  {
546  raw_fd_ostream file(FD, false);
547 
548  // Get the data and its length
549  Expected<StringRef> BufOrErr = C.getBuffer();
550  failIfError(BufOrErr.takeError());
551  StringRef Data = BufOrErr.get();
552 
553  // Write the data.
554  file.write(Data.data(), Data.size());
555  }
556 
557  // If we're supposed to retain the original modification times, etc. do so
558  // now.
559  if (OriginalDates) {
560  auto ModTimeOrErr = C.getLastModified();
561  failIfError(ModTimeOrErr.takeError());
562  failIfError(sys::fs::setLastAccessAndModificationTime(FD, ModTimeOrErr.get()));
563  }
564 
565  if (close(FD)) {
566  fail("Could not close the file");
567  }
568 }

588 static void performReadOperation(ArchiveOperation Operation, object::Archive *OldArchive) {
589  if (Operation == Extract && OldArchive->isThin()) {
590  fail("extracting from a thin archive is not supported");
591  }
592 
593  bool Filter = !Members.empty();
594  StringMap<int> MemberCount;
595  {
596  Error Err = Error::success();
597  for (auto &C : OldArchive->children(Err)) {
598  Expected<StringRef> NameOrErr = C.getName();
599  failIfError(NameOrErr.takeError());
600  StringRef Name = NameOrErr.get();
601 
602  if (Filter) {
603  auto I = find_if(Members, [Name](StringRef Path) {
604  return comparePaths(Name, Path);
605  });
606  if (I == Members.end()) {
607  continue;
608  }
609  if (CountParam && ++MemberCount[Name] != CountParam) {
610  continue;
611  }
612  Members.erase(I);
613  }
614 
615  switch (Operation) {
616  default:
617  llvm_unreachable("Not a read operation");
618  case Print:
619  doPrint(Name, C);
620  break;
621  case DisplayTable:
622  doDisplayTable(Name, C);
623  break;
624  case Extract:
625  doExtract(Name, C);
626  break;
627  }
628  }
629  failIfError(std::move(Err));
630  }
631 
632  if (Members.empty()) {
633  return;
634  }
635  for (StringRef Name : Members) {
636  WithColor::error(errs(), ToolName) << "'" << Name << "' was not found\n";
637  }
638  exit(1);
639 }

641 static void addChildMember(std::vector<NewArchiveMember> &Members,
642  const object::Archive::Child &M,
643  bool FlattenArchive = false) {
644  if (Thin && !M.getParent()->isThin()) {
645  fail("cannot convert a regular archive to a thin one");
646  }
647  Expected<NewArchiveMember> NMOrErr = NewArchiveMember::getOldMember(M, Deterministic);
648  failIfError(NMOrErr.takeError());
649  // If the child member we're trying to add is thin, use the path relative to
650  // the archive it's in, so the file resolves correctly.
651  if (Thin && FlattenArchive) {
652  StringSaver Saver(Alloc);
653  Expected<std::string> FileNameOrErr(M.getName());
654  failIfError(FileNameOrErr.takeError());
655  if (sys::path::is_absolute(*FileNameOrErr)) {
656  NMOrErr->MemberName = Saver.save(sys::path::convert_to_slash(*FileNameOrErr));
657  } else {
658  FileNameOrErr = M.getFullName();
659  failIfError(FileNameOrErr.takeError());
660  Expected<std::string> PathOrErr = computeArchiveRelativePath(ArchiveName, *FileNameOrErr);
661  NMOrErr->MemberName = Saver.save(PathOrErr ? *PathOrErr : sys::path::convert_to_slash(*FileNameOrErr));
662  }
663  }
664  if (FlattenArchive && identify_magic(NMOrErr->Buf->getBuffer()) == file_magic::archive) {
665  Expected<std::string> FileNameOrErr = M.getFullName();
666  failIfError(FileNameOrErr.takeError());
667  object::Archive &Lib = readLibrary(*FileNameOrErr);
668  // When creating thin archives, only flatten if the member is also thin.
669  if (!Thin || Lib.isThin()) {
670  Error Err = Error::success();
671  // Only Thin archives are recursively flattened.
672  for (auto &Child : Lib.children(Err)) {
673  addChildMember(Members, Child, /*FlattenArchive=*/Thin);
674  }
675  failIfError(std::move(Err));
676  return;
677  }
678  }
679  Members.push_back(std::move(*NMOrErr));
680 }

682 static void addMember(std::vector<NewArchiveMember> &Members, StringRef FileName, bool FlattenArchive = false) {
683  Expected<NewArchiveMember> NMOrErr = NewArchiveMember::getFile(FileName, Deterministic);
684  failIfError(NMOrErr.takeError(), FileName);
685  StringSaver Saver(Alloc);
686  // For regular archives, use the basename of the object path for the member
687  // name. For thin archives, use the full relative paths so the file resolves
688  // correctly.
689  if (!Thin) {
690  NMOrErr->MemberName = sys::path::filename(NMOrErr->MemberName);
691  } else {
692  if (sys::path::is_absolute(FileName)) {

720 static InsertAction computeInsertAction(ArchiveOperation Operation,
721  const object::Archive::Child &Member,
722  StringRef Name,
723  std::vector<StringRef>::iterator &Pos,
724  StringMap<int> &MemberCount) {
725  if (Operation == QuickAppend || Members.empty()) {
726  return IA_AddOldMember;
727  }
728  auto MI = find_if(Members, [Name](StringRef Path) { return comparePaths(Name, Path); });
729 
730  if (MI == Members.end()) {
731  return IA_AddOldMember;
732  }
733 
734  Pos = MI;
735 
736  if (Operation == Delete) {
737  if (CountParam && ++MemberCount[Name] != CountParam) {
738  return IA_AddOldMember;
739  }
740  return IA_Delete;
741  }
742 
743  if (Operation == Move) {
744  return IA_MoveOldMember;
745  }
746 
747  if (Operation == ReplaceOrInsert) {
748  if (!OnlyUpdate) {
749  if (RelPos.empty()) {
750  return IA_AddNewMember;
751  }
752  return IA_MoveNewMember;
753  }
754 
755  // We could try to optimize this to a fstat, but it is not a common
756  // operation.
757  sys::fs::file_status Status;
758  failIfError(sys::fs::status(*MI, Status), *MI);
759  auto ModTimeOrErr = Member.getLastModified();
760  failIfError(ModTimeOrErr.takeError());
761  if (Status.getLastModificationTime() < ModTimeOrErr.get()) {
762  if (RelPos.empty()) {
763  return IA_AddOldMember;
764  }
765  return IA_MoveOldMember;
766  }
767 
768  if (RelPos.empty()) {
769  return IA_AddNewMember;
770  }
771  return IA_MoveNewMember;
772  }
773  llvm_unreachable("No such operation");
774 }

776 // We have to walk this twice and computing it is not trivial, so creating an
777 // explicit std::vector is actually fairly efficient.
778 static std::vector<NewArchiveMember> computeNewArchiveMembers(ArchiveOperation Operation, object::Archive *OldArchive) {
779  std::vector<NewArchiveMember> Ret;
780  std::vector<NewArchiveMember> Moved;
781  int InsertPos = -1;
782  if (OldArchive) {
783  Error Err = Error::success();
784  StringMap<int> MemberCount;
785  for (auto &Child : OldArchive->children(Err)) {
786  int Pos = Ret.size();
787  Expected<StringRef> NameOrErr = Child.getName();
788  failIfError(NameOrErr.takeError());
789  std::string Name = std::string(NameOrErr.get());
790  if (comparePaths(Name, RelPos)) {
791  assert(AddAfter || AddBefore);
792  if (AddBefore) {
793  InsertPos = Pos;
794  } else {
795  InsertPos = Pos + 1;
796  }
797  }
798 
799  std::vector<StringRef>::iterator MemberI = Members.end();
800  InsertAction Action = computeInsertAction(Operation, Child, Name, MemberI, MemberCount);
801  switch (Action) {
802  case IA_AddOldMember:
803  addChildMember(Ret, Child, /*FlattenArchive=*/Thin);
804  break;
805  case IA_AddNewMember:
806  addMember(Ret, *MemberI);
807  break;
808  case IA_Delete:
809  break;
810  case IA_MoveOldMember:
811  addChildMember(Moved, Child, /*FlattenArchive=*/Thin);
812  break;
813  case IA_MoveNewMember:
814  addMember(Moved, *MemberI);
815  break;
816  }
817  // When processing elements with the count param, we need to preserve the
818  // full members list when iterating over all archive members. For
819  // instance, "llvm-ar dN 2 archive.a member.o" should delete the second
820  // file named member.o it sees; we are not done with member.o the first
821  // time we see it in the archive.
822  if (MemberI != Members.end() && !CountParam) {
823  Members.erase(MemberI);
824  }
825  }
826  failIfError(std::move(Err));
827  }
828 
829  if (Operation == Delete) {
830  return Ret;
831  }
832 
833  if (!RelPos.empty() && InsertPos == -1) {
834  fail("insertion point not found");
835  }
836 
837  if (RelPos.empty()) {
838  InsertPos = Ret.size();
839  }
840 
841  assert(unsigned(InsertPos) <= Ret.size());
842  int Pos = InsertPos;
843  for (auto &M : Moved) {
844  Ret.insert(Ret.begin() + Pos, std::move(M));
845  ++Pos;
846  }
847 
848  if (AddLibrary) {
849  assert(Operation == QuickAppend);
850  for (auto &Member : Members) {
851  addMember(Ret, Member, /*FlattenArchive=*/true);
852  }
853  return Ret;
854  }
855 
856  std::vector<NewArchiveMember> NewMembers;
857  for (auto &Member : Members) {
858  addMember(NewMembers, Member, /*FlattenArchive=*/Thin);
859  }
860  Ret.reserve(Ret.size() + NewMembers.size());
861  std::move(NewMembers.begin(), NewMembers.end(), std::inserter(Ret, std::next(Ret.begin(), InsertPos)));
862 
863  return Ret;
864 }
