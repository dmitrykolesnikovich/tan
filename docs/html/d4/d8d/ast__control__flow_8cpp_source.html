13  if (!condition) {

15  report_code_error(condition_token, "Invalid condition expression " + condition_token->to_string());
16  }
17 
18  /// convert to bool if not
19  condition = TypeSystem::ConvertTo(cs, cs->get_builder()->getInt1Ty(), condition, false);

21  /// Create blocks for the then (and else) clause
22  Function *func = cs->get_builder()->GetInsertBlock()->getParent();
23  BasicBlock *then_bb = BasicBlock::Create(*cs->get_context(), "then", func);
24  BasicBlock *else_bb = BasicBlock::Create(*cs->get_context(), "else");

26

32  if (!cs->get_builder()->GetInsertBlock()->back().isTerminator()) { cs->get_builder()->CreateBr(merge_bb); }
33 
34  /// emit else block
35  func->getBasicBlockList().push_back(else_bb);
36  cs->get_builder()->SetInsertPoint(else_bb);
37  if (_has_else) { _children[2]->codegen(cs); }
38  /// create a br instruction if there is no terminator instruction at the end of this block
39  if (!cs->get_builder()->GetInsertBlock()->back().isTerminator()) { cs->get_builder()->CreateBr(merge_bb); }
40 
41  /// emit merge block
42  func->getBasicBlockList().push_back(merge_bb);
43  cs->get_builder()->SetInsertPoint(merge_bb);

52 llvm::Value *ASTBreakContinue::codegen(CompilerSession *cs) {

54  if (!loop) { report_code_error(_token, "Any break/continue statement must be inside loop"); }
55  auto s = loop->get_loop_start();
56  auto e = loop->get_loop_end();
57  if (_type == ASTType::BREAK) {
58  cs->get_builder()->CreateBr(e);
