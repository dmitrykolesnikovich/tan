85 size_t ASTArrayLiteral::nud() {
86  _end_index = _start_index + 1; /// skip '['
87  if (_parser->at(_end_index)->value == "]") { /// empty array
88  ++_end_index;
89  return _end_index;
90  }
91  ASTType element_type = ASTType::INVALID;
92  while (!_parser->eof(_end_index)) {
93  if (_parser->at(_end_index)->value == ",") {
94  ++_end_index;
95  continue;
96  } else if (_parser->at(_end_index)->value == "]") {
97  ++_end_index;
98  break;
99  }
100  auto node = _parser->peek(_end_index);
101  if (!node) { report_code_error(_token, "Unexpected token"); }
102  /// check whether element types are the same
103  if (element_type == ASTType::INVALID) {
104  element_type = node->_type;
105  } else {
106  if (element_type != node->_type) {
107  report_code_error(_token, "All elements in an array must have the same type");
108  }
109  }
110  if (is_ast_type_in(node->_type, TypeSystem::LiteralTypes)) {
111  if (node->_type == ASTType::ARRAY_LITERAL) { ++_end_index; }
112  _end_index = node->parse(_parser, _cs);
113  _children.push_back(node);
114  } else {
115  report_code_error(_token, "Expect literals");
116  }
117  }
118  return _end_index;
119 }
