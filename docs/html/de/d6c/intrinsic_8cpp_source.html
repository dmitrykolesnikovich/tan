34 Value *Intrinsic::codegen(CompilerSession *cs) {
35  ASTNodePtr tmp = nullptr;
36  switch (_intrinsic_type) {
37  case IntrinsicType::FILENAME:
38  tmp = std::make_shared<ASTStringLiteral>(_parser->get_filename(), _start_index);
39  _llvm_value = tmp->codegen(cs);
40  break;
41  case IntrinsicType::LINENO:
42  tmp = std::make_shared<ASTNumberLiteral>(_token->l, _start_index);
43  _llvm_value = tmp->codegen(cs);
44  break;
45  case IntrinsicType::GET_DECL:
46  tmp = std::make_shared<ASTStringLiteral>(cs->get(_str_data)->get_src(), _start_index);
47  _llvm_value = tmp->codegen(cs);
48  break;
49  case IntrinsicType::STACK_TRACE: {
50  TAN_ASSERT(_children[0]->_type == ASTType::FUNC_CALL);
51  TAN_ASSERT(_children[0]->_children[0]->_type == ASTType::NUM_LITERAL);
52  auto arg = ast_cast<ASTNumberLiteral>(_children[0]->_children[0]);
53  TAN_ASSERT(!arg->is_float());
54  _llvm_value = codegen_get_stack_trace(cs, (size_t) (arg->_ivalue));
55  break;
56  }
57  default:
58  TAN_ASSERT(_children.size());
59  _llvm_value = _children[0]->codegen(cs);
60  break;
61  }
62  return _llvm_value;
63 }

81 void Intrinsic::determine_type() {
82  auto *token = _parser->at(_start_index);
83  if (Intrinsic::intrinsics.find(token->value) == Intrinsic::intrinsics.end()) {
84  report_code_error(_token, "Invalid intrinsic");
85  }
86  _intrinsic_type = Intrinsic::intrinsics[token->value];
87  ASTNodePtr underlying_ast = nullptr;
88  switch (_intrinsic_type) {
89  case IntrinsicType::NOOP:
90  _is_typed = false;
91  underlying_ast = std::make_shared<ASTFunctionCall>(token, _start_index);
92  _ty = nullptr;
93  break;
94  case IntrinsicType::STACK_TRACE:
95  // TODO: set _tyty
96  _is_lvalue = true;
97  underlying_ast = std::make_shared<ASTFunctionCall>(token, _start_index);
98  break;
99  case IntrinsicType::LINENO:
100  _ty = ASTTy::Create(TY_OR3(Ty::INT, Ty::BIT32, Ty::UNSIGNED), std::vector<ASTNodePtr>());
101  break;
102  case IntrinsicType::FILENAME:
103  _ty = ASTTy::Create(Ty::STRING, std::vector<ASTNodePtr>());
104  break;
105  case IntrinsicType::GET_DECL:
106  _ty = ASTTy::Create(Ty::STRING, std::vector<ASTNodePtr>());
107  parse_get_decl();
108  break;
109  default:
110  report_code_error(token, "Unknown intrinsic");
111  }
112  if (underlying_ast) {
113  _lbp = underlying_ast->_lbp;
114  _rbp = underlying_ast->_rbp;
115  _children.push_back(underlying_ast);
116  }
117 }

168 static void init_abort(CompilerSession *cs) {
169  if (Intrinsic::assert_initialized) { return; }
170  Function *assert_func = cs->get_module()->getFunction("__assert");
171  { /// fn __assert(msg: char*, file: char*, line: int) : void;
172  if (!assert_func) {
173  Type *ret_type = cs->get_builder()->getVoidTy();
174  std::vector<Type *> arg_types
175  {cs->get_builder()->getInt8PtrTy(), cs->get_builder()->getInt8PtrTy(), cs->get_builder()->getInt32Ty(),};
176  FunctionType *FT = FunctionType::get(ret_type, arg_types, false);
177  assert_func = Function::Create(FT, Function::ExternalWeakLinkage, "__assert", cs->get_module().get());
178  }
179  }
180  { /// fn assert(a: i1) : void;
181  Function *func = cs->get_module()->getFunction("assert");
182  if (!func) {
183  Type *ret_type = cs->get_builder()->getVoidTy();
184  std::vector<Type *> arg_types{cs->get_builder()->getInt1Ty()};
185  FunctionType *func_t = FunctionType::get(ret_type, arg_types, false);
186  func = Function::Create(func_t, Function::ExternalLinkage, "assert", cs->get_module().get());
187  /// body
188  BasicBlock *main_block = BasicBlock::Create(*cs->get_context(), "func_entry", func);
189  cs->get_builder()->SetInsertPoint(main_block);
190  Value *arg = &(*func->args().begin());
191  Value *z = ConstantInt::get(cs->get_builder()->getInt1Ty(), 0, false);
192  Value *condition = cs->get_builder()->CreateICmpEQ(z, arg);
193 
194  BasicBlock *then_bb = BasicBlock::Create(*cs->get_context(), "then", func);
195  BasicBlock *merge_bb = BasicBlock::Create(*cs->get_context(), "fi");
196 
197  cs->get_builder()->CreateCondBr(condition, then_bb, merge_bb);
198  cs->get_builder()->SetInsertPoint(then_bb);
199  auto *st = codegen_get_stack_trace(cs, 1);
200  auto *int_t = cs->get_builder()->getInt32Ty();
201  auto *zero = ConstantInt::get(int_t, 0);
202  auto *one = ConstantInt::get(int_t, 1);
203  auto *two = ConstantInt::get(int_t, 2);
204  std::vector<Value *> args = {cs->get_builder()->CreateLoad(cs->get_builder()->CreateGEP(st, {zero, one})),
205  cs->get_builder()->CreateLoad(cs->get_builder()->CreateGEP(st, {zero, zero})),
206  cs->get_builder()->CreateLoad(cs->get_builder()->CreateGEP(st, {zero, two}))};
207  cs->get_builder()->CreateCall(assert_func, args);
208  cs->get_builder()->CreateBr(merge_bb);
209  func->getBasicBlockList().push_back(merge_bb);
210  cs->get_builder()->SetInsertPoint(merge_bb);
211  cs->get_builder()->CreateRetVoid();
212  }
213  }
214  Intrinsic::assert_initialized = true;
215 }
