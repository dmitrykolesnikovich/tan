81 void Intrinsic::resolve() {
82  auto *token = _parser->at(_start_index);
83  if (Intrinsic::intrinsics.find(token->value) == Intrinsic::intrinsics.end()) {
84  report_code_error(_token, "Invalid intrinsic");
85  }
86  _intrinsic_type = Intrinsic::intrinsics[token->value];
87  ASTNodePtr underlying_ast = nullptr;
88  switch (_intrinsic_type) {
89  case IntrinsicType::NOOP:
90  _is_typed = false;
91  underlying_ast = std::make_shared<ASTFunctionCall>(token, _start_index);
92  _ty = nullptr;
93  break;
94  case IntrinsicType::ABORT:
95  _is_typed = false;
96  underlying_ast = std::make_shared<ASTFunctionCall>(token, _start_index);
97  _ty = nullptr;
98  break;
99  case IntrinsicType::STACK_TRACE:
100  underlying_ast = std::make_shared<ASTFunctionCall>(token, _start_index);
101  break;
102  case IntrinsicType::LINENO:
103  _ty = ASTTy::Create(TY_OR3(Ty::INT, Ty::BIT32, Ty::UNSIGNED), vector<ASTNodePtr>());
104  break;
105  case IntrinsicType::FILENAME:
106  _ty = ASTTy::Create(Ty::STRING, vector<ASTNodePtr>());
107  break;
108  case IntrinsicType::GET_DECL:
109  _ty = ASTTy::Create(Ty::STRING, vector<ASTNodePtr>());
110  parse_get_decl();
111  break;
112  case IntrinsicType::COMP_PRINT:
113  underlying_ast = std::make_shared<ASTFunctionCall>(token, _start_index);
114  break;
115  default:
116  report_code_error(token, "Unknown intrinsic");
117  }
118  if (underlying_ast) {
119  _lbp = underlying_ast->_lbp;
120  _rbp = underlying_ast->_rbp;
121  _children.push_back(underlying_ast);
122  }
123  _parsed = true;
124 }
