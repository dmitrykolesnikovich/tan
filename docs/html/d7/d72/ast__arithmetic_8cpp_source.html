38  } else {

40  }
41 
42  if (lhs->getType()->isFloatingPointTy()) {

46  } else if (_type == ASTType::DIVIDE) {

52  }
53  }

57  } else if (_type == ASTType::DIVIDE) {

65  } else if (_type == ASTType::SUBTRACT) {

67  }
68  TAN_ASSERT(false);
