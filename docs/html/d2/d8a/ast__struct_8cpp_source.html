8 size_t ASTStruct::nud() {
9  _end_index = _start_index + 1; /// skip "struct"
10  /// struct typename
11  auto id = _parser->parse<ASTType::ID>(_end_index, true);
12  TAN_ASSERT(id->is_named());
13  _type_name = id->get_name();
14  _children.push_back(id);
15  _cs->add(id->get_name(), this->shared_from_this()); /// add self to current scope
16 
17  /// struct body
18  auto comp_stmt = _parser->peek(_end_index);
19  _end_index = comp_stmt->parse(_parser, _cs); // TODO: parse forward declaration
20  auto members = comp_stmt->_children;
21 
22  /// resolve member names and types
23  ASTNodePtr var_decl = nullptr;
24  size_t n = comp_stmt->_children.size();
25  _member_names.reserve(n);
26  _children.reserve(n);
27  for (size_t i = 0; i < n; ++i) {
28  if (members[i]->_type == ASTType::VAR_DECL) { /// member variable without initial value
29  var_decl = members[i];
30  _children.push_back(var_decl->get_ty());
31  } else if (members[i]->_type == ASTType::ASSIGN) { /// member variable with an initial value
32  var_decl = members[i]->_children[0];
33  auto initial_value = members[i]->_children[1];
34  if (!is_ast_type_in(initial_value->_type, TypeSystem::LiteralTypes)) {
35  report_code_error(_token, "Invalid initial value of the member variable");
36  }
37  _children.push_back(initial_value->get_ty()); /// initial value is set to ASTTy in ASTLiteral::get_ty()
38  } else { report_code_error(_token, "Invalid struct member"); }
39  auto name = var_decl->get_name();
40  _member_names.push_back(name);
41  _member_indices[name] = i;
42  }
43  this->resolve();
44  return _end_index;
45 }
