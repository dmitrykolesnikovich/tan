18

21  _end_index = comp_stmt->parse(_parser, _cs); // TODO: parse forward declaration
22  auto members = comp_stmt->_children;
23 
24  /// resolve member names and types
25  ASTNodePtr var_decl = nullptr;
26  size_t n = comp_stmt->_children.size();
27  _member_names.reserve(n);
28  _children.reserve(n);
29  for (size_t i = 0; i < n; ++i) {
30  if (members[i]->_type == ASTType::VAR_DECL) { /// member variable without initial value
31  var_decl = members[i];
32  _children.push_back(var_decl->get_ty());
33  } else if (members[i]->_type == ASTType::ASSIGN) { /// member variable with an initial value
34  var_decl = members[i]->_children[0];
35  auto initial_value = members[i]->_children[1];
36  if (!is_ast_type_in(initial_value->_type, TypeSystem::LiteralTypes)) {
37  report_code_error(_token, "Invalid initial value of the member variable");
38  }
39  _children.push_back(initial_value->get_ty()); /// initial value is set to ASTTy in ASTLiteral::get_ty()
40  } else { report_code_error(_token, "Invalid struct member"); }
41  auto name = var_decl->get_name();
42  _member_names.push_back(name);
43  _member_indices[name] = i;
44  }
