9 llvm::Value *ASTLoop::codegen(CompilerSession *cs) {
10  cs->set_current_debug_location(_token->l, _token->c);
11  cs->set_current_loop(this->shared_from_this());
12  if (_loop_type == ASTLoopType::WHILE) {
13  /*
14  * Results should like this:
15  *
16  * ...
17  * loop:
18  * exit condition check, goto 'loop_body' or 'after_loop'
19  * loop_body:
20  * ...
21  * goto 'loop'
22  * after_loop:
23  * ...
24  * */
25 
26  Function *func = cs->get_builder()->GetInsertBlock()->getParent();
27 
28  /// make sure to set _loop_start and _loop_end before generating loop_body, cuz break and continue statements
29  /// use these two (get_loop_start() and get_loop_end())
30  _loop_start = BasicBlock::Create(*cs->get_context(), "loop", func);
31  BasicBlock *loop_body = BasicBlock::Create(*cs->get_context(), "loop_body", func);
32  _loop_end = BasicBlock::Create(*cs->get_context(), "after_loop", func);
33 
34  /// start loop
35  // create a br instruction if there is no terminator instruction at the end of this block
36  if (!cs->get_builder()->GetInsertBlock()->back().isTerminator()) { cs->get_builder()->CreateBr(_loop_start); }
37 
38  /// exit condition
39  cs->get_builder()->SetInsertPoint(_loop_start);
40  auto *cond = _children[0]->codegen(cs);
41  cond = TypeSystem::ConvertTo(cs, cs->get_builder()->getInt1Ty(), cond, false);
42  cs->get_builder()->CreateCondBr(cond, loop_body, _loop_end);
43 
44  /// loop body
45  cs->get_builder()->SetInsertPoint(loop_body);
46  _children[1]->codegen(cs);
47 
48  /// go back to the start of the loop
49  // create a br instruction if there is no terminator instruction at the end of this block
50  if (!cs->get_builder()->GetInsertBlock()->back().isTerminator()) { cs->get_builder()->CreateBr(_loop_start); }
51 
52  /// end loop
53  cs->get_builder()->SetInsertPoint(_loop_end);
54  } else { TAN_ASSERT(false); }
55  cs->set_current_loop(nullptr);
56  return nullptr;
57 }
