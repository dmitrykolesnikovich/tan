68 ASTNodePtr Parser::peek(size_t &index) {
69  if (index >= _tokens.size()) { return nullptr; }
70  Token *token = _tokens[index];
71  /// skip comments
72  while (token->type == TokenType::COMMENTS) {
73  ++index;
74  token = _tokens[index];
75  }
76  ASTNodePtr node;
77  if (token->value == "@") { /// intrinsics
78  node = std::make_shared<Intrinsic>(token, index);
79  } else if (token->value == "+" && token->type == TokenType::BOP) {
80  node = std::make_shared<ASTArithmetic>(ASTType::SUM, token, index);
81  } else if (token->value == "-" && token->type == TokenType::BOP) {
82  node = std::make_shared<ASTArithmetic>(ASTType::SUBTRACT, token, index);
83  } else if (token->value == "*" && token->type == TokenType::BOP) {
84  node = std::make_shared<ASTArithmetic>(ASTType::MULTIPLY, token, index);
85  } else if (token->value == "/" && token->type == TokenType::BOP) {
86  node = std::make_shared<ASTArithmetic>(ASTType::DIVIDE, token, index);
87  } else if (token->value == "=" && token->type == TokenType::BOP) {
88  node = std::make_shared<ASTAssignment>(token, index);
89  } else if (token->value == "!" || token->value == "~") {
90  node = std::make_shared<ASTNot>(token, index);
91  } else if (token->value == "[") {
92  auto prev = this->at(index - 1);
93  if (prev->type != TokenType::ID && prev->value != "]" && prev->value != ")") {
94  /// array literal if there is no identifier, "]", or ")" before
95  node = std::make_shared<ASTArrayLiteral>(token, index);
96  } else {
97  /// otherwise bracket access
98  node = std::make_shared<ASTMemberAccess>(token, index);
99  }
100  } else if (token->type == TokenType::RELOP) {
101  if (token->value == ">") {
102  node = std::make_shared<ASTCompare>(ASTType::GT, token, index);
103  } else if (token->value == ">=") {
104  node = std::make_shared<ASTCompare>(ASTType::GE, token, index);
105  } else if (token->value == "<") {
106  node = std::make_shared<ASTCompare>(ASTType::LT, token, index);
107  } else if (token->value == "<=") {
108  node = std::make_shared<ASTCompare>(ASTType::LE, token, index);
109  } else if (token->value == "==") {
110  node = std::make_shared<ASTCompare>(ASTType::EQ, token, index);
111  } else if (token->value == "!=") {
112  node = std::make_shared<ASTCompare>(ASTType::NE, token, index);
113  }
114  } else if (token->type == TokenType::INT) {
115  node = std::make_shared<ASTNumberLiteral>(token->value, false, token, index);
116  } else if (token->type == TokenType::FLOAT) {
117  node = std::make_shared<ASTNumberLiteral>(token->value, true, token, index);
118  } else if (token->type == TokenType::STRING) {
119  node = std::make_shared<ASTStringLiteral>(token, index);
120  } else if (token->type == TokenType::CHAR) {
121  node = std::make_shared<ASTCharLiteral>(token, index);
122  } else if (token->type == TokenType::ID) {
123  Token *next = _tokens[index + 1];
124  if (next->value == "(") { node = std::make_shared<ASTFunctionCall>(token, index); }
125  else { node = std::make_shared<ASTIdentifier>(token, index); }
126  } else if (token->type == TokenType::PUNCTUATION && token->value == "(") {
127  node = std::make_shared<ASTParenthesis>(token, index);
128  } else if (token->type == TokenType::KEYWORD) { /// keywords
129  node = peek_keyword(token, index);
130  } else if (token->type == TokenType::BOP && token->value == ".") { /// member access
131  node = std::make_shared<ASTMemberAccess>(token, index);
132  } else if (check_typename_token(token)) { /// types
133  node = std::make_shared<ASTTy>(token, index);
134  } else if (token->value == "&") {
135  node = std::make_shared<ASTAmpersand>(token, index);
136  } else if (token->type == TokenType::PUNCTUATION && token->value == "{") {
137  node = std::make_shared<ASTStatement>(true, token, index);
138  } else if (check_terminal_token(token)) { /// this MUST be the last thing to check
139  return nullptr;
140  } else {
141  report_code_error(token, "Unknown token " + token->to_string());
142  }
143  return node;
144 }

146 ASTNodePtr Parser::next_expression(size_t &index, int rbp) {
147  ASTNodePtr node = peek(index);
148  ++index;
149  if (!node) { return nullptr; }
150  auto n = node;
151  index = n->parse(this, _cs);
152  auto left = n;
153  node = peek(index);
154  if (!node) { return left; }
155  while (rbp < node->_lbp) {
156  node = peek(index);
157  n = node;
158  index = n->parse(left, this, _cs);
159  left = n;
160  node = peek(index);
161  if (!node) { break; }
162  }
163  return left;
164 }
