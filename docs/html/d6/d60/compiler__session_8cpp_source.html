62 void CompilerSession::set(const std::string &name, ASTNodePtr value) {
63  // search from the outer-est scope to the inner-est scope
64  auto scope = _scope.end(); // scope is an iterator
65  --scope;
66  while (scope >= _scope.begin()) {
67  auto search = (*scope)->_named.find(name);
68  if (search != (*scope)->_named.end()) {
69  throw std::runtime_error("Cannot set the value of " + name);
70  }
71  --scope;
72  }
73  (*scope)->_named[name] = value;
74 }

154 void CompilerSession::emit_object(const std::string &filename) {
155  _di_builder->finalize(); /// important: do this before any pass
156 
157  /// run function pass on all functions in the current module
158  for (auto &f: *_module.get()) {
159  if (f.getName() == "main") { /// special case for main function
160  // mark as used, prevent LLVM from deleting it
161  llvm::appendToUsed(*_module.get(), {(GlobalValue *) &f});
162  }
163  _fpm->run(f);
164  }
165  _fpm->doFinalization();
166 
167  /// generate object files
168  std::error_code ec;
169  llvm::raw_fd_ostream dest(filename, ec, llvm::sys::fs::OF_None);
170  if (ec) { throw std::runtime_error("Could not open file: " + ec.message()); }
171 
172  auto file_type = llvm::CGFT_ObjectFile;
173  if (_target_machine->addPassesToEmitFile(*_mpm.get(), dest, nullptr, file_type)) {
174  throw std::runtime_error("Target machine can't emit a file of this type");
175  }
176  _mpm->run(*_module);
177  dest.flush();
178 }
